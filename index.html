Бэкенд (Node.js) Создай файл `server.js`: ```js const express = require('express'); const Stripe = require('stripe'); const app = express(); const stripe = Stripe('sk_test_YOUR_SECRET_KEY'); app.use(express.static('.')); app.use(express.json()); app.post('/create-checkout-session', async (req, res) => { const session = await stripe.checkout.sessions.create({ payment_method_types: ['card'], line_items: [{ price_data: { currency: 'eur', product_data: { name: 'Таро PRO доступ' }, unit_amount: 500 }, quantity: 1 }], mode: 'payment', success_url: 'http://localhost:3000/success', cancel_url: 'http://localhost:3000/cancel' }); res.json({ sessionId: session.id, publicKey: 'pk_test_YOUR_PUBLIC_KEY' }); }); app.listen(3000, () => console.log('Server running')); ``` --- # 🧠 3.
